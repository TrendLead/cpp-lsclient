196 size_t LSClient::streamCallbackWrapper(void* ptr, size_t size, size_t nmemb, void* obj) {
197 
202  LSClient* lsc = static_cast<LSClient*>(obj);
203 
204  if (ptr != NULL) {
205  std::string ls_stream(static_cast<const char*>(ptr), size * nmemb);
206 
207  if (lsc->getStatus() == LS_STATUS_CONNECTED ||
208  lsc->getStatus() == LS_STATUS_RECEIVING ) {
209 
210  //debug
211  //cout << ls_stream << endl;
212 
213  //actually parses data comming in stream connection
214  if ( ls_stream.find("|") != std::string::npos ) {
215 
216  lsc->setStatus(LS_STATUS_RECEIVING);
217 
218  std::vector<std::string> values = split(ls_stream,'|');
219  std::vector<std::string> values_map = split(values[0],',');
220 
221  int tnum = atoi(values_map[0].c_str());
222  int indices_num = atoi(values_map[1].c_str()) - 1;
223 
224  values.erase(values.begin());
225  LSTable::append(tnum, indices_num, values);
226 
227 
228  }
229 
230  }
231 
232  std::vector<std::string> resdata = split(ls_stream,'\n');
233  for (int i=0;i<resdata.size();i++) {
234  trim(resdata[i]);
235  }
236 
237  if ( resdata[0] == "OK" ) {
238  lsc->setStatus(LS_STATUS_CONNECTED);
239  //parse Data
240  for (int i=1;i<resdata.size();i++) {
241  std::vector<std::string> parsed_line = split(resdata[i],':');
242 
243  if (parsed_line.size() > 1) {
244  if (parsed_line[0] == "SessionId") {
245  lsc->setSessionId(parsed_line[1]);
246  }
247  else if (parsed_line[0] == "ControlAddress") {
248  lsc->setControlEndpoint(parsed_line[1]);
249  }
250  }
251  }
252  }
253 
254  }
255  return size*nmemb;
256 }
